## Week 17：RBAC权限管理

### 🔐 Day 1-2：RBAC基础概念

#### RBAC核心组件学习
```yaml
# 我的RBAC学习笔记
# 1. ServiceAccount - 身份标识
apiVersion: v1
kind: ServiceAccount
metadata:
  name: app-service-account
  namespace: development
  labels:
    app: myapp
    environment: dev
automountServiceAccountToken: false  # 手动控制是否自动挂载token

---
# 2. Role - 命名空间级别权限
apiVersion: rbac.authorization.k8s.io/v1
kind: Role
metadata:
  name: app-developer-role
  namespace: development
rules:
# Pod权限
- apiGroups: [""]
  resources: ["pods", "pods/log", "pods/exec"]
  verbs: ["get", "list", "watch", "create", "delete"]
  
# ConfigMap权限
- apiGroups: [""]
  resources: ["configmaps", "secrets"]
  verbs: ["get", "list", "watch", "create", "update", "patch"]
  
# Deployment权限
- apiGroups: ["apps"]
  resources: ["deployments", "deployments/scale"]
  verbs: ["get", "list", "watch", "create", "update", "patch"]
  
# 服务权限
- apiGroups: [""]
  resources: ["services"]
  verbs: ["get", "list", "watch", "create", "update", "patch"]
  
# 详细权限控制
- apiGroups: [""]
  resources: ["pods"]
  verbs: ["get", "list"]
  resourceNames: ["app-pod-*"]  # 只允许访问特定名称的资源
  
# 带条件的权限
- apiGroups: [""]
  resources: ["pods"]
  verbs: ["delete"]
  resourceNames: ["app-pod-*"]
  # 只能删除带有特定标签的Pod

---
# 3. ClusterRole - 集群级别权限
apiVersion: rbac.authorization.k8s.io/v1
kind: ClusterRole
metadata:
  name: cluster-reader
rules:
# 只读权限
- apiGroups: [""]
  resources: ["nodes", "persistentvolumes", "namespaces"]
  verbs: ["get", "list", "watch"]
  
# 集群状态查看
- apiGroups: ["metrics.k8s.io"]
  resources: ["nodes", "pods"]
  verbs: ["get", "list"]
  
# 事件查看
- apiGroups: [""]
  resources: ["events"]
  verbs: ["get", "list", "watch"]

---
# 4. RoleBinding - 命名空间级别绑定
apiVersion: rbac.authorization.k8s.io/v1
kind: RoleBinding
metadata:
  name: app-developer-binding
  namespace: development
subjects:
# 绑定到用户
- kind: User
  name: alice@example.com
  apiGroup: rbac.authorization.k8s.io
  
# 绑定到组
- kind: Group
  name: developers
  apiGroup: rbac.authorization.k8s.io
  
# 绑定到ServiceAccount
- kind: ServiceAccount
  name: app-service-account
  namespace: development
  
roleRef:
  kind: Role
  name: app-developer-role
  apiGroup: rbac.authorization.k8s.io

---
# 5. ClusterRoleBinding - 集群级别绑定
apiVersion: rbac.authorization.k8s.io/v1
kind: ClusterRoleBinding
metadata:
  name: cluster-admin-binding
subjects:
- kind: User
  name: admin@example.com
  apiGroup: rbac.authorization.k8s.io
roleRef:
  kind: ClusterRole
  name: cluster-admin
  apiGroup: rbac.authorization.k8s.io
```

#### RBAC权限测试
```bash
# 我的RBAC实践记录

# 1. 创建测试ServiceAccount
kubectl create serviceaccount test-user -n development

# 2. 创建权限测试Role
kubectl apply -f rbac-test-role.yaml

# 3. 创建RoleBinding
kubectl apply -f rbac-test-binding.yaml

# 4. 测试权限（使用ServiceAccount）
kubectl auth can-i get pods -n development --as=system:serviceaccount:development:test-user
# 应该返回 yes

kubectl auth can-i delete pods -n development --as=system:serviceaccount:development:test-user
# 应该返回 no

# 5. 查看详细权限
kubectl auth can-i --list -n development --as=system:serviceaccount:development:test-user

# 6. 测试跨命名空间权限
kubectl auth can-i get pods -n production --as=system:serviceaccount:development:test-user
# 应该返回 no

# 7. 模拟真实用户权限测试
kubectl auth can-i get secrets -n development --as=alice@example.com
kubectl auth can-i create deployments -n development --as=alice@example.com

# 8. 权限调试
kubectl auth can-i '*' '*' -n development --as=alice@example.com
# 查看用户是否有所有权限
```

#### Pod安全上下文配置
```yaml
# 我的Pod安全学习笔记
apiVersion: v1
kind: Pod
metadata:
  name: security-context-demo
spec:
  # 1. 安全策略
  securityContext:
    runAsNonRoot: true  # 必须以非root用户运行
    runAsUser: 1000     # 指定用户ID
    runAsGroup: 1000    # 指定组ID
    fsGroup: 1000       # 卷挂载的组ID
    supplementalGroups: [1001, 1002]  # 补充组ID
    sysctls:
    - name: net.core.somaxconn
      value: "1024"
    - name: kernel.msgmax
      value: "65536"
  
  containers:
  - name: secure-app
    image: nginx:1.23
    
    # 2. 容器安全上下文
    securityContext:
      # 权限设置
      allowPrivilegeEscalation: false  # 禁止权限提升
      readOnlyRootFilesystem: true     # 只读根文件系统
      privileged: false               # 非特权容器
      
      # 能力管理
      capabilities:
        drop:                         # 删除所有能力
        - ALL
        add:                          # 只添加必需的能力
        - NET_BIND_SERVICE            # 绑定低端口
        - CHOWN                       # 改变文件所有者
      
      # SELinux选项
      seLinuxOptions:
        level: "s0:c123,c456"
      
      # AppArmor配置
      apparmorProfile: runtime/default
      
      # seccomp配置
      seccompProfile:
        type: RuntimeDefault
    
    # 3. 资源限制
    resources:
      requests:
        memory: "128Mi"
        cpu: "100m"
      limits:
        memory: "256Mi"
        cpu: "200m"
    
    volumeMounts:
    - name: tmp-volume
      mountPath: /tmp
    - name: cache-volume
      mountPath: /var/cache/nginx
    - name: config-volume
      mountPath: /etc/nginx/nginx.conf
      subPath: nginx.conf
      readOnly: true
  
  volumes:
  - name: tmp-volume
    emptyDir: {}
  - name: cache-volume
    emptyDir: {}
  - name: config-volume
    configMap:
      name: nginx-config
      defaultMode: 0400  # 只读权限
```

#### NetworkPolicy网络隔离
```yaml
# 我的NetworkPolicy学习笔记
# 1. 默认拒绝所有入站流量
apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: default-deny-ingress
  namespace: development
spec:
  podSelector: {}  # 选择所有Pod
  policyTypes:
  - Ingress
  # 没有定义ingress规则，表示拒绝所有入站流量

---
# 2. 默认拒绝所有出站流量
apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: default-deny-egress
  namespace: development
spec:
  podSelector: {}
  policyTypes:
  - Egress

---
# 3. 允许特定Pod间通信
apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: allow-app-to-db
  namespace: development
spec:
  podSelector:
    matchLabels:
      app: database  # 目标Pod标签
  policyTypes:
  - Ingress
  ingress:
  - from:
    - podSelector:
        matchLabels:
          app: backend  # 源Pod标签
    ports:
    - protocol: TCP
      port: 5432  # PostgreSQL端口

---
# 4. 允许访问特定命名空间
apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: allow-namespace-traffic
  namespace: development
spec:
  podSelector: {}
  policyTypes:
  - Ingress
  ingress:
  - from:
    - namespaceSelector:
        matchLabels:
          name: monitoring  # 允许来自monitoring命名空间的流量

---
# 5. 允许特定IP段访问
apiVersion: networking.k8s.io/v1

  name: allow-external-ip
```

```yaml
    - alert: PodRestartingTooOften
      expr: rate(kube_pod_container_status_restarts_total[15m]) > 0.1
      for: 5m
      labels:
        severity: warning
        team: frontend
      annotations:
        summary: "Pod {{ $labels.pod }} is restarting too often"
        description: "Pod {{ $labels.pod }} in namespace {{ $labels.namespace }} has restarted {{ $value }} times in the last 15 minutes"

    - alert: HighCPUUsage
      expr: rate(container_cpu_usage_seconds_total{container!=""}[5m]) > 0.8
      for: 10m

        severity: critical
        team: frontend
      annotations:
        summary: "High CPU usage detected"
        description: "Container {{ $labels.container }} in pod {{ $labels.pod }} is using {{ $value | humanizePercentage }} of CPU"
    
    # 内存使用率告警

      expr: container_memory_usage_bytes{container!=""} / container_spec_memory_limit_bytes > 0.9

      labels:
        severity: warning

      annotations:

        team: frontend
```

#### 应用指标暴露
```yaml
# 我的应用指标暴露学习
# 1. 在应用中暴露Prometheus指标
apiVersion: v1
kind: ConfigMap
metadata:
  name: app-metrics-config
data:
  app.py: |
    from flask import Flask, Response
    from prometheus_client import Counter, Histogram, Gauge, generate_latest
    import time
    import random
    
    app = Flask(__name__)
    
    # 定义指标
    request_count = Counter('app_requests_total', 'Total requests', ['method', 'endpoint'])
    request_duration = Histogram('app_request_duration_seconds', 'Request duration', ['method', 'endpoint'])
    active_users = Gauge('app_active_users', 'Number of active users')
    error_count = Counter('app_errors_total', 'Total errors', ['type'])
    
    @app.before_request
    def before_request():
        request.start_time = time.time()
    
    @app.after_request
    def after_request(response):
        # 记录请求数
        request_count.labels(method=request.method, endpoint=request.endpoint).inc()
        
        # 记录请求时长
        duration = time.time() - request.start_time
        request_duration.labels(method=request.method, endpoint=request.endpoint).observe(duration)
        
        return response
    
    @app.route('/')
    def hello():
        # 模拟随机错误
        if random.random() < 0.1:
            error_count.labels(type='random_error').inc()
            return "Error!", 500
        
        # 更新活跃用户指标
        active_users.set(random.randint(1, 100))
        
        return "Hello World!"
    
    @app.route('/metrics')
    def metrics():
        return Response(generate_latest(), mimetype='text/plain')
    
    if __name__ == '__main__':
        app.run(host='0.0.0.0', port=8080)

---
# 2. 部署带指标的应用
apiVersion: apps/v1
kind: Deployment
metadata:
  name: metrics-app

  replicas: 3
  selector:
    matchLabels:
      app: metrics-app
  template:

      labels:
        app: metrics-app
      annotations:
        prometheus.io/scrape: "true"  # Prometheus会自动发现
        prometheus.io/port: "8080"
        prometheus.io/path: "/metrics"
    spec:
      containers:
      - name: app
        image: myapp:metrics
        ports:
        - containerPort: 8080
          name: web
        - containerPort: 8080
          name: metrics
        env:
        - name: METRICS_ENABLED
          value: "true"
        - name: METRICS_PORT
          value: "8080"
        resources:
          requests:
            memory: "128Mi"
            cpu: "100m"

            memory: "256Mi"
```

```yaml
---
# 2. Fluentd配置文件
apiVersion: v1
kind: ConfigMap
metadata:
  name: fluentd-config
  namespace: logging
data:
  fluent.conf: |
    # 系统日志
    <source>
      @type tail
      @id in_tail_container_logs
      path /var/log/containers/*.log
      pos_file /var/log/fluentd-containers.log.pos
      tag "kubernetes.*"
      read_from_head true
      <parse>
        @type multi_format
        <pattern>
          format json
          time_key time
          time_format %Y-%m-%dT%H:%M:%S.%NZ
        </pattern>
        <pattern>
          format /^(?<time>.+) (?<stream>stdout|stderr) [^ ]* (?<message>.*)$/
          time_format %Y-%m-%dT%H:%M:%S.%N%:z
        </pattern>
      </parse>
    </source>
    
    # Kubernetes元数据过滤
    <filter kubernetes.**>
      @type kubernetes_metadata
      @id filter_kube_metadata
      kubernetes_url "https://#{ENV['FLUENT_KUBERNETES_HOSTNAME']}:443"
      verify_ssl false
      skip_labels false
      skip_container_metadata false
    </filter>
    
    # 日志级别解析
    <filter kubernetes.**>
      @type grep
      <regexp>
        key log
        pattern /^(?<timestamp>\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}) (?<level>\w+) (?<message>.*)$/
      </regexp>
    </filter>
    
    # 日志格式化
    <filter kubernetes.**>
      @type record_transformer
      <record>
```
